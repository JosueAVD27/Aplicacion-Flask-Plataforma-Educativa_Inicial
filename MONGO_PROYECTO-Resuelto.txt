#################################################
Consulta: Encontrar roles con sus permisos

db.Roles.aggregate([
	{$lookup:
		{
			from: "Permisos", 
			localField: "permiso", 
			foreignField: "permiso", 
			as: "Permiso"
		}
	}, {$unwind: "$Permiso"},
	{$match: { rol: "Administrador"}}
]).pretty()
